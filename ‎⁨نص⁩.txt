  <script>
    let score1 = 0, score2 = 0;

    function startGame() {
      const name1 = document.getElementById('t1-input').value || 'گرووپی ۱';
      const name2 = document.getElementById('t2-input').value || 'گرووپی ۲';
      
      document.getElementById('name1').innerText = name1;
      document.getElementById('name2').innerText = name2;
      
      document.getElementById('setup-screen').classList.add('hidden');
      document.getElementById('game-screen').classList.remove('hidden');
    }

    function updateScore(team, value) {
      if (team === 1) {
        score1 = Math.max(0, score1 + value);
        document.getElementById('s1').innerText = score1;
      } else {
        score2 = Math.max(0, score2 + value);
        document.getElementById('s2').innerText = score2;
      }
    }

    function resetGame() {
      score1 = 0; score2 = 0;
      document.getElementById('s1').innerText = 0;
      document.getElementById('s2').innerText = 0;
      document.getElementById('game-screen').classList.add('hidden');
      document.getElementById('setup-screen').classList.remove('hidden');
    }
  </script>
